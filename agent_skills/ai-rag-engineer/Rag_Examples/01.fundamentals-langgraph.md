# LangGraph Fundamentals para Agentes RAG

Este skill proporciona las bases sólidas para construir **agentes y flujos de trabajo** confiables usando **LangGraph**.

## ¿Cuándo usar este skill?

Utiliza este skill cuando necesites construir:

- Flujos multi-paso de análisis complejo (documentos técnicos, contratos, reportes, investigaciones)
- Agentes que requieren **decisiones condicionales** basadas en reglas de dominio
- Sistemas con **alta trazabilidad** y control de estado
- Bucles de refinamiento con verificación humana o automática
- Arquitecturas donde la precisión y la explicabilidad son críticas

def node_validar_normativa(state: LegalDocumentState) -> dict:

    Valida que las normas citadas estén vigentes y sean aplicables.
    Integra con base de conocimiento normativa (RAG).
    """
    from proyectos.Rag_Legal.services.legal_validator import validate_norm_vigency
    
    norms = state.get('applicable_norms', [])
    if not norms:
        return {'validation_errors': [{'type': 'no_norms', 'message': 'No se identificaron normas aplicables'}]}
    
    validation_results = []
    for norm in norms:
        result = validate_norm_vigency(norm)  # Consulta a vector store normativo
        validation_results.append({
            'norma': norm,
            'vigente': result['vigente'],
            'ultima_actualizacion': result.get('fecha_actualizacion'),
            'observaciones': result.get('observaciones')
        })
    
    # Identificar normas no vigentes como error
    errors = [
        {'type': 'norma_no_vigente', 'norma': r['norma'], 'mensaje': 'La norma no está vigente'}
        for r in validation_results if not r['vigente']
    ]
    
    return {
        'norm_vigency': {r['norma']: {'vigente': r['vigente'], 'fecha_actualizacion': r['ultima_actualizacion']} 
                        for r in validation_results},
        'validation_errors': errors,
        'requires_human_review': len(errors) > 0
    }

from langgraph.graph import StateGraph, START, END
from langgraph.checkpoint.postgres import PostgresSaver  # Producción
from contextlib import contextmanager

@contextmanager
def get_legal_graph_connection():
    """Context manager para conexión a base de checkpoints."""
    if settings.USE_POSTGRES_CHECKPOINTS:
        from psycopg import Connection
        with Connection.connect(settings.DATABASE_URL) as conn:
            yield PostgresSaver(conn)
    else:
        yield MemorySaver()

def build_tutela_analyzer_graph() -> CompiledGraph:
    """
    Construye el grafo completo para análisis de tutelas.
    Incluye: clasificación, extracción, análisis, validación, generación.
    """
    builder = create_legal_graph_builder(AnalisisTutelaState)
    
    # === Nodes ===
    builder.add_node("clasificar", node_clasificar_documento)
    builder.add_node("extraer_hechos", node_extraer_hechos_tutela)
    builder.add_node("identificar_derechos", node_identificar_derechos_fundamentales)
    builder.add_node("verificar_procedencia", node_verificar_procedencia_tutela)
    builder.add_node("generar_recomendaciones", node_generar_recomendaciones_tutela)
    builder.add_node("validar_citas", node_validar_normativa)
    builder.add_node("formatear_respuesta", node_formatear_respuesta_legal)
    builder.add_node("revision_humana", node_preparar_para_revision_humana)
    
    # === Edges ===
    builder.add_edge(START, "clasificar")
    
    # Router inicial
    builder.add_conditional_edges(
        "clasificar",
        lambda s: "tutela" if s.get('document_type') == 'tutela' else "no_tutela",
        {
            "tutela": "extraer_hechos",
            "no_tutela": "revision_humana"
        }
    )
    
    # Pipeline de análisis de tutela
    builder.add_edge("extraer_hechos", "identificar_derechos")
    builder.add_edge("identificar_derechos", "verificar_procedencia")
    
    # Router de procedencia
    builder.add_conditional_edges(
        "verificar_procedencia",
        lambda s: "procedente" if s.get('mecanismo_procedente') else "improcedente",
        {
            "procedente": "generar_recomendaciones",
            "improcedente": "revision_humana"
        }
    )
    
    # Validación y formato
    builder.add_edge("generar_recomendaciones", "validar_citas")
    builder.add_edge("validar_citas", "formatear_respuesta")
    builder.add_edge("formatear_respuesta", END)
    
    # Fallback a revisión humana
    builder.add_edge("revision_humana", END)
    
    # === Compilación con producción settings ===
    with get_legal_graph_connection() as checkpointer:
        return builder.compile(
            checkpointer=checkpointer,
            interrupt_before=["revision_humana"] if settings.ENABLE_HUMAN_IN_LOOP else None,
            debug=settings.DEBUG_MODE
        )

# Instancia singleton para reutilización
_tutela_graph = None

def get_tutela_analyzer() -> CompiledGraph:
    """Obtiene instancia del grafo (singleton con lazy initialization)."""
    global _tutela_graph
    if _tutela_graph is None:
        _tutela_graph = build_tutela_analyzer_graph()
    return _tutela_graph


Ejecución con Manejo de Errores

async def analyze_tutela(
    document_content: str, 
    document_id: str,
    timeout_seconds: int = 120
) -> dict:
    """
    Ejecuta el análisis de tutela con manejo robusto de errores.
    
    Args:
        document_content: Texto completo de la tutela
        document_id: Identificador único para trazabilidad
        timeout_seconds: Timeout máximo de ejecución
        
    Returns:
        dict: Resultado del análisis o error detallado
    """
    import asyncio
    from langgraph.errors import GraphRecursionError
    
    graph = get_tutela_analyzer()
    
    initial_state = AnalisisTutelaState(
        document_id=document_id,
        raw_content=document_content,
        current_step="init",
        retry_count=0,
        max_retries=2,
        confidence_scores={},
        execution_log=[],
        errors=[],
        requires_human_review=False
    )
    
    try:
        # Ejecución con timeout
        result = await asyncio.wait_for(
            graph.ainvoke(initial_state, config={"configurable": {"thread_id": document_id}}),
            timeout=timeout_seconds
        )
        
        # Post-procesamiento de resultado
        return {
            'success': True,
            'document_id': document_id,
            'analysis': {
                'summary': result.get('analysis_summary'),
                'recommendations': result.get('recommendations', []),
                'citations': result.get('citations', []),
                'risks': result.get('risks', [])
            },
            'metadata': {
                'execution_log': result.get('execution_log', []),
                'confidence': result.get('confidence_scores'),
                'requires_review': result.get('requires_human_review', False)
            }
        }
        
    except asyncio.TimeoutError:
        logger.error(f"[analyze_tutela] Timeout para doc_id={document_id}")
        return {
            'success': False,
            'error': 'timeout',
            'message': f'El análisis excedió el tiempo límite de {timeout_seconds}s',
            'document_id': document_id,
            'requires_human_review': True
        }
        
    except GraphRecursionError:
        logger.error(f"[analyze_tutela] Recursión excesiva para doc_id={document_id}")
        return {
            'success': False,
            'error': 'recursion_limit',
            'message': 'El grafo excedió el límite de iteraciones',
            'document_id': document_id,
            'requires_human_review': True
        }
        
    except Exception as e:
        logger.exception(f"[analyze_tutela] Error inesperado: {str(e)}")
        return {
            'success': False,
            'error': 'internal_error',
            'message': f'Error interno: {str(e)}',
            'document_id': document_id,
            'requires_human_review': True
        }

# proyectos/Rag_Legal/metrics/legal_agent_metrics.py

from dataclasses import dataclass
from typing import Dict, List
import time

@dataclass
class AgentExecutionMetrics:
    document_id: str
    start_time: float
    end_time: float
    nodes_executed: List[str]
    confidence_scores: Dict[str, float]
    errors: List[str]
    requires_human_review: bool
    
    @property
    def execution_time_seconds(self) -> float:
        return self.end_time - self.start_time
    
    @property
    def average_confidence(self) -> float:
        if not self.confidence_scores:
            return 0.0
        return sum(self.confidence_scores.values()) / len(self.confidence_scores)
    
    @property
    def quality_score(self) -> float:
        """Score compuesto: 0-100"""
        # Factores ponderados
        time_score = max(0, 100 - (self.execution_time_seconds / 2))  # <10s = 100pts
        confidence_score = self.average_confidence * 40  # Máx 40pts
        error_penalty = len(self.errors) * 15  # -15pts por error
        review_bonus = 20 if not self.requires_human_review else 0
        
        return min(100, max(0, time_score + confidence_score + review_bonus - error_penalty))

class LegalAgentMetricsCollector:
    """Colector de métricas para agentes legales."""
    
    def __init__(self, metrics_backend="memory"):
        self.metrics: List[AgentExecutionMetrics] = []
        self.backend = metrics_backend  # memory, postgres, prometheus
        
    def record_execution(self, metrics: AgentExecutionMetrics):
        """Registra una ejecución para análisis posterior."""
        self.metrics.append(metrics)
        
        if self.backend == "prometheus":
            self._export_to_prometheus(metrics)
    
    def get_summary_stats(self, last_n_hours: int = 24) -> dict:
        """Genera estadísticas resumidas para dashboard."""
        cutoff = time.time() - (last_n_hours * 3600)
        recent = [m for m in self.metrics if m.start_time >= cutoff]
        
        if not recent:
            return {"count": 0}
        
        return {
            "total_executions": len(recent),
            "avg_execution_time": sum(m.execution_time_seconds for m in recent) / len(recent),
            "avg_quality_score": sum(m.quality_score for m in recent) / len(recent),
            "success_rate": sum(1 for m in recent if not m.errors) / len(recent),
            "human_review_rate": sum(1 for m in recent if m.requires_human_review) / len(recent),
            "avg_confidence": sum(m.average_confidence for m in recent) / len(recent),
        }

class RAGEnhancedLegalAgent:
    """
    Agente legal que combina LangGraph para flujo de trabajo 
    con RAG para recuperación de normativa contextual.
    """
    
    def __init__(self, graph: CompiledGraph, rag_retriever):
        self.graph = graph
        self.rag_retriever = rag_retriever
        
    async def invoke_with_rag(self, state: LegalDocumentState) -> dict:
        """
        Ejecuta el grafo enriqueciendo el contexto con RAG cuando sea necesario.
        """
        # Hook pre-ejecución: enriquecer estado con contexto normativo
        if state.get('applicable_norms'):
            enriched_context = await self._fetch_normative_context(
                norms=state['applicable_norms'],
                jurisdiction=state.get('jurisdiction')
            )
            state['rag_context'] = enriched_context
        
        # Ejecutar grafo principal
        result = await self.graph.ainvoke(state)
        
        # Hook post-ejecución: validar citas contra RAG
        if result.get('citations'):
            validated_citations = await self._validate_citations_with_rag(
                citations=result['citations']
            )
            result['citations'] = validated_citations
        
        return result
    
    async def _fetch_normative_context(self, norms: List[str], jurisdiction: str) -> str:
        """Recupera contexto normativo relevante desde el vector store."""
        queries = [f"{norma} {jurisdiction or ''}".strip() for norma in norms]
        
        results = []
        for query in queries:
            docs = self.rag_retriever.get_relevant_documents(query, k=3)
            results.append("\n".join([d.page_content for d in docs]))
        
        return "\n\n---\n\n".join(results)

        "status": "accepted",
        "document_id": request.document_id,
        "estimated_completion": "2-3 minutos"
    }

async def process_tutela_analysis(
    document_content: str,
    document_id: str,
    callback_url: str = None
):
    """Task en background para procesamiento asíncrono."""
    from proyectos.Rag_Legal.metrics import LegalAgentMetricsCollector
    
    start_time = time.time()
    collector = LegalAgentMetricsCollector()
    
    try:
        # Ejecutar análisis
        result = await analyze_tutela(
            document_content=document_content,
            document_id=document_id
        )
        
        # Registrar métricas
        metrics = AgentExecutionMetrics(
            document_id=document_id,
            start_time=start_time,
            end_time=time.time(),
            nodes_executed=result.get('execution_log', []),
            confidence_scores=result.get('metadata', {}).get('confidence', {}),
            errors=result.get('errors', []),
            requires_human_review=result.get('requires_human_review', False)
        )
        collector.record_execution(metrics)
        
        # Callback opcional
        if callback_url and result['success']:
            await send_webhook(callback_url, result)
            
    except Exception as e:
        logger.exception(f"Error en process_tutela_analysis: {e}")
        # Registrar métrica de error
        collector.record_execution(AgentExecutionMetrics(
            document_id=document_id,
            start_time=start_time,
            end_time=time.time(),
            nodes_executed=[],
            confidence_scores={},
            errors=[str(e)],
            requires_human_review=True
        ))
